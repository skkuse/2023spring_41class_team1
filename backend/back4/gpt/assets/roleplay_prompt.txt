#include<stdio.h>

// Calculating factorial
int factorialcalculation(int n) {
if(n == 0)
    return 1;
else
{
    int fact = 1;
for(int i = 1; i <= n; i++) {
fact *= i;
}
return fact;
}
}

// Main function
int main() {
    int n;
    printf("Enter a number: ");
    scanf("%d", &n);

if(n < 0)
printf("Factorial of negative numbers doesn't exist.\n");
else {
    int result = factorialcalculation(n);
    printf("Factorial of %d is %d\n", n, result);
}
return 0;
}
